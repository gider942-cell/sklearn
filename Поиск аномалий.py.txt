import numpy as np 
import matplotlib.pyplot as plt 
from sklearn.cluster import DBSCAN
from sklearn.datasets import make_blobs 
from sklearn.preprocessing import StandardScaler

X, y_true = make_blobs(
    n_samples = 300,
    centers = 4,
    cluster_std = 0.6,
    random_state = 42
)

np.random.seed(42)
n_outliers = 20
outliers = np.random.uniform(low=-5, high=5, size=(n_outliers, 2))
X = np.vstack([X, outliers])

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

eps = 0.5 
min_samples = 5 

db = DBSCAN(eps=eps, min_samples=min_samples)
labels = db.fit_predict(X_scaled)

anomaly_mask = labels == -1
n_anomalies = np.sum(anomaly_mask)
anomaly_ratio = n_anomalies / len(labels)

print(f"Число аномалий: {n_anomalies}")
print(f"Доля аномалий: {anomaly_ratio:.3f}")

plt.figure(figsize=(8, 6))
plt.scatter(
    X[~anomaly_mask, 0], X[~anomaly_mask, 1],
    c=labels[~anomaly_mask], cmap='viridis', s=30, alpha=0.8, label='Кластеры'
)
plt.scatter(
    X[anomaly_mask, 0], X[anomaly_mask, 1],
    c='red', s=40, edgecolors='k', label='Аномалии (шум)'
)
plt.title(f"DBSCAN: аномалии (eps={eps}, min_samples={min_samples})")
plt.xlabel("Признак 1")
plt.ylabel("Признак 2")
plt.legend()
plt.grid(True, linestyle="--", alpha=0.3)
plt.tight_layout()
plt.show()
