from sklearn.datasets import load_iris 
from sklearn.model_selection import train_test_split 
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score, classification_report
import numpy as np 

iris = load_iris()
X = iris.data 
y = iris.target

X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42, stratify=y
)

k = 5 
model = KNeighborsClassifier(n_neighbors=k)
model.fit(X_train, y_train)

y_pred = model.predict(X_test)

accuracy = accuracy_score(y_test, y_pred)
print(f"Точность модели на тесте: {accuracy:.3f}")
print("\nОтчёт по классам:")
print(classification_report(y_test, y_pred, target_names=iris.target_names))

new_samples = np.array([
    [5.1, 3.5, 1.4, 0.2],
    [6.7, 3.0, 5.2, 2.3]
])
predictions = model.predict(new_samples)
predicted_species = [iris.target_names[p] for p in predictions]

print("\nПредсказания для новых образцов:")
for sample, species in zip(new_samples, predicted_species):
    print(f"{sample} -> {species}")
